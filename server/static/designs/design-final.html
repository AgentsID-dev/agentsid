Works in your language

import { AgentsID } from '@agentsid/sdk';

// Initialize the client
const aid = new AgentsID({
  projectId: process.env.AGENTSID_PROJECT,
});

// Register an agent with scoped permissions
const agent = await aid.register({
  name: 'recon-agent',
  role: 'analyst',
  tools: ['search.web', 'search.docs'],
  delegatedBy: 'user_01HXYZ',
});

// Validate before every tool call
const allowed = await agent.authorize('search.web', {
  query: 'latest CVEs',
});

if (allowed.granted) {
  const result = await searchWeb('latest CVEs');
}

import com.agentsid.sdk.AgentsID;
import com.agentsid.sdk.Agent;

// Initialize the client
AgentsID aid = new AgentsID(
    System.getenv("AGENTSID_PROJECT")
);

// Register an agent with scoped permissions
Agent agent = aid.register(AgentConfig.builder()
    .name("recon-agent")
    .role("analyst")
    .tools(List.of("search.web", "search.docs"))
    .delegatedBy("user_01HXYZ")
    .build());

// Validate before every tool call
AuthResult allowed = agent.authorize(
    "search.web",
    Map.of("query", "latest CVEs")
);

if (allowed.isGranted()) {
    Result result = searchWeb("latest CVEs");
}
